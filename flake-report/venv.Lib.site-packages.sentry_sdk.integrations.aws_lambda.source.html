287 def get_lambda_bootstrap():
288 # type: () -> Optional[Any]
289  
290 # Python 2.7: Everything is in `__main__`.
291 #
292 # Python 3.7: If the bootstrap module is *already imported*, it is the
293 # one we actually want to use (no idea what's in __main__)
294 #
295 # Python 3.8: bootstrap is also importable, but will be the same file
296 # as __main__ imported under a different name:
297 #
298 # sys.modules['__main__'].__file__ == sys.modules['bootstrap'].__file__
299 # sys.modules['__main__'] is not sys.modules['bootstrap']
300 #
301 # Python 3.9: bootstrap is in __main__.awslambdaricmain
302 #
303 # On container builds using the `aws-lambda-python-runtime-interface-client`
304 # (awslamdaric) module, bootstrap is located in sys.modules['__main__'].bootstrap
305 #
306 # Such a setup would then make all monkeypatches useless.
307 if "bootstrap" in sys.modules:
308 return sys.modules["bootstrap"]
309 elif "__main__" in sys.modules:
310 module = sys.modules["__main__"]
311 # python3.9 runtime
312 if hasattr(module, "awslambdaricmain") and hasattr(
313 module.awslambdaricmain, "bootstrap"
314 ):
315 return module.awslambdaricmain.bootstrap
316 elif hasattr(module, "bootstrap"):
317 # awslambdaric python module in container builds
318 return module.bootstrap
319  
320 # python3.8 runtime
321 return module
322 else:
323 return None
