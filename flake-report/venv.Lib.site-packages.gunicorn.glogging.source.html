119 def parse_syslog_address(addr):
120  
121 # unix domain socket type depends on backend
122 # SysLogHandler will try both when given None
123 if addr.startswith("unix://"):
124 sock_type = None
125  
126 # set socket type only if explicitly requested
127 parts = addr.split("#", 1)
128 if len(parts) == 2:
129 addr = parts[0]
130 if parts[1] == "dgram":
131 sock_type = socket.SOCK_DGRAM
132  
133 return (sock_type, addr.split("unix://")[1])
134  
135 if addr.startswith("udp://"):
136 addr = addr.split("udp://")[1]
137 socktype = socket.SOCK_DGRAM
138 elif addr.startswith("tcp://"):
139 addr = addr.split("tcp://")[1]
140 socktype = socket.SOCK_STREAM
141 else:
142 raise RuntimeError("invalid syslog address")
143  
144 if '[' in addr and ']' in addr:
145 host = addr.split(']')[0][1:].lower()
146 elif ':' in addr:
147 host = addr.split(':')[0].lower()
148 elif addr == "":
149 host = "localhost"
150 else:
151 host = addr.lower()
152  
153 addr = addr.split(']')[-1]
154 if ":" in addr:
155 port = addr.split(':', 1)[1]
156 if not port.isdigit():
157 raise RuntimeError("%r is not a valid port number." % port)
158 port = int(port)
159 else:
160 port = 514
161  
162 return (socktype, (host, port))

291 def atoms(self, resp, req, environ, request_time):
292 """ Gets atoms for log formatting.
293 """
294 status = resp.status
295 if isinstance(status, str):
296 status = status.split(None, 1)[0]
297 atoms = {
298 'h': environ.get('REMOTE_ADDR', '-'),
299 'l': '-',
300 'u': self._get_user(environ) or '-',
301 't': self.now(),
302 'r': "%s %s %s" % (environ['REQUEST_METHOD'],
303 environ['RAW_URI'],
304 environ["SERVER_PROTOCOL"]),
305 's': status,
306 'm': environ.get('REQUEST_METHOD'),
307 'U': environ.get('PATH_INFO'),
308 'q': environ.get('QUERY_STRING'),
309 'H': environ.get('SERVER_PROTOCOL'),
310 'b': getattr(resp, 'sent', None) is not None and str(resp.sent) or '-',
311 'B': getattr(resp, 'sent', None),
312 'f': environ.get('HTTP_REFERER', '-'),
313 'a': environ.get('HTTP_USER_AGENT', '-'),
314 'T': request_time.seconds,
315 'D': (request_time.seconds * 1000000) + request_time.microseconds,
316 'M': (request_time.seconds * 1000) + int(request_time.microseconds / 1000),
317 'L': "%d.%06d" % (request_time.seconds, request_time.microseconds),
318 'p': "<%s>" % os.getpid()
319 }
320  
321 # add request headers
322 if hasattr(req, 'headers'):
323 req_headers = req.headers
324 else:
325 req_headers = req
326  
327 if hasattr(req_headers, "items"):
328 req_headers = req_headers.items()
329  
330 atoms.update({"{%s}i" % k.lower(): v for k, v in req_headers})
331  
332 resp_headers = resp.headers
333 if hasattr(resp_headers, "items"):
334 resp_headers = resp_headers.items()
335  
336 # add response headers
337 atoms.update({"{%s}o" % k.lower(): v for k, v in resp_headers})
338  
339 # add environ variables
340 environ_variables = environ.items()
341 atoms.update({"{%s}e" % k.lower(): v for k, v in environ_variables})
342  
343 return atoms

410 def _set_handler(self, log, output, fmt, stream=None):
411 # remove previous gunicorn log handler
412 h = self._get_gunicorn_handler(log)
413 if h:
414 log.handlers.remove(h)
415  
416 if output is not None:
417 if output == "-":
418 h = logging.StreamHandler(stream)
419 else:
420 util.check_is_writable(output)
421 h = logging.FileHandler(output)
422 # make sure the user can reopen the file
423 try:
424 os.chown(h.baseFilename, self.cfg.user, self.cfg.group)
425 except OSError:
426 # it's probably OK there, we assume the user has given
427 # /dev/null as a parameter.
428 pass
429  
430 h.setFormatter(fmt)
431 h._gunicorn = True
432 log.addHandler(h)

434 def _set_syslog_handler(self, log, cfg, fmt, name):
435 # setup format
436 prefix = cfg.syslog_prefix or cfg.proc_name.replace(":", ".")
437  
438 prefix = "gunicorn.%s.%s" % (prefix, name)
439  
440 # set format
441 fmt = logging.Formatter(r"%s: %s" % (prefix, fmt))
442  
443 # syslog facility
444 try:
445 facility = SYSLOG_FACILITIES[cfg.syslog_facility.lower()]
446 except KeyError:
447 raise RuntimeError("unknown facility name")
448  
449 # parse syslog address
450 socktype, addr = parse_syslog_address(cfg.syslog_addr)
451  
452 # finally setup the syslog handler
453 h = logging.handlers.SysLogHandler(address=addr,
454 facility=facility, socktype=socktype)
455  
456 h.setFormatter(fmt)
457 h._gunicorn = True
458 log.addHandler(h)

460 def _get_user(self, environ):
461 user = None
462 http_auth = environ.get("HTTP_AUTHORIZATION")
463 if http_auth and http_auth.lower().startswith('basic'):
464 auth = http_auth.split(" ", 1)
465 if len(auth) == 2:
466 try:
467 # b64decode doesn't accept unicode in Python < 3.3
468 # so we need to convert it to a byte string
469 auth = base64.b64decode(auth[1].strip().encode('utf-8'))
470 # b64decode returns a byte string
471 user = auth.split(b":", 1)[0].decode("UTF-8")
472 except (TypeError, binascii.Error, UnicodeDecodeError) as exc:
473 self.debug("Couldn't get username: %s", exc)
474 return user
