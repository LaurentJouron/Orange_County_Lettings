160 def raise_exception(client):
161 # type: (Optional[Client]) -> None
162 """
163 Raise an exception. If the client is not in the hub, rebind it.
164 """
165 hub = Hub.current
166 if hub.client is None:
167 hub.bind_client(client)
168 exc_info = sys.exc_info()
169 with capture_internal_exceptions():
170 _capture_exception(exc_info, hub)
171 reraise(*exc_info)
