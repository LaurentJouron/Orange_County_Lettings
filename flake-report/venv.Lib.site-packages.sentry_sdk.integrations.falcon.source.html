173 def sentry_patched_handle_exception(self, *args):
174 # type: (falcon.API, *Any) -> Any
175 # NOTE(jmagnusson): falcon 2.0 changed falcon.API._handle_exception
176 # method signature from `(ex, req, resp, params)` to
177 # `(req, resp, ex, params)`
178 ex = response = None
179 with capture_internal_exceptions():
180 ex = next(argument for argument in args if isinstance(argument, Exception))
181 response = next(
182 argument for argument in args if isinstance(argument, falcon.Response)
183 )
184  
185 was_handled = original_handle_exception(self, *args)
186  
187 if ex is None or response is None:
188 # Both ex and response should have a non-None value at this point; otherwise,
189 # there is an error with the SDK that will have been captured in the
190 # capture_internal_exceptions block above.
191 return was_handled

202 client_options=client.options,

236 def _exception_leads_to_http_5xx(ex, response):
237 # type: (Exception, falcon.Response) -> bool
238 is_server_error = isinstance(ex, falcon.HTTPError) and (ex.status or "").startswith(
239 "5"
240 )
241 is_unhandled_error = not isinstance(
242 ex, (falcon.HTTPError, falcon.http_status.HTTPStatus)
243 )
244  
245 # We only check the HTTP status on Falcon 3 because in Falcon 2, the status on the response
246 # at the stage where we capture it is listed as 200, even though we would expect to see a 500
247 # status. Since at the time of this change, Falcon 2 is ca. 4 years old, we have decided to
248 # only perform this check on Falcon 3+, despite the risk that some handled errors might be
249 # reported to Sentry as unhandled on Falcon 2.
250 return (is_server_error or is_unhandled_error) and (
251 not FALCON3 or _has_http_5xx_status(response)
252 )
