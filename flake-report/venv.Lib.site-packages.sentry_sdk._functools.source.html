80 def update_wrapper(
81 wrapper, wrapped, assigned=WRAPPER_ASSIGNMENTS, updated=WRAPPER_UPDATES
82 ):
83 # type: (Any, Any, Any, Any) -> Any
84 """Update a wrapper function to look like the wrapped function
85  
86 wrapper is the function to be updated
87 wrapped is the original function
88 assigned is a tuple naming the attributes assigned directly
89 from the wrapped function to the wrapper function (defaults to
90 functools.WRAPPER_ASSIGNMENTS)
91 updated is a tuple naming the attributes of the wrapper that
92 are updated with the corresponding attribute from the wrapped
93 function (defaults to functools.WRAPPER_UPDATES)
94 """
95 for attr in assigned:
96 try:
97 value = getattr(wrapped, attr)
98 except AttributeError:
99 pass
100 else:
101 setattr(wrapper, attr, value)
102 for attr in updated:
103 getattr(wrapper, attr).update(getattr(wrapped, attr, {}))
104 # Issue #17482: set __wrapped__ last so we don't inadvertently copy it
105 # from the wrapped function when updating __dict__
106 wrapper.__wrapped__ = wrapped
107 # Return the wrapper so this can be used as a decorator via partial()
108 return wrapper

111 def wraps(wrapped, assigned=WRAPPER_ASSIGNMENTS, updated=WRAPPER_UPDATES):
112 # type: (Callable[..., Any], Any, Any) -> Callable[[Callable[..., Any]], Callable[..., Any]]
113 """Decorator factory to apply update_wrapper() to a wrapper function
114  
115 Returns a decorator that invokes update_wrapper() with the decorated
116 function as the wrapper argument and the arguments to wraps() as the
117 remaining arguments. Default arguments are as for update_wrapper().
118 This is a convenience function to simplify applying partial() to
119 update_wrapper().
120 """
121 return partial(update_wrapper, wrapped=wrapped, assigned=assigned, updated=updated)
