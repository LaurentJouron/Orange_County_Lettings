45 def _strip_pii(command):
46 # type: (Dict[str, Any]) -> Dict[str, Any]
47 for key in command:
48 is_safe_field = key in SAFE_COMMAND_ATTRIBUTES
49 if is_safe_field:
50 # Skip if safe key
51 continue
52  
53 update_db_command = key == "update" and "findAndModify" not in command
54 if update_db_command:
55 # Also skip "update" db command because it is save.
56 # There is also an "update" key in the "findAndModify" command, which is NOT safe!
57 continue
58  
59 # Special stripping for documents
60 is_document = key == "documents"
61 if is_document:
62 for doc in command[key]:
63 for doc_key in doc:
64 doc[doc_key] = "%s"
65 continue
66  
67 # Special stripping for dict style fields
68 is_dict_field = key in ["filter", "query", "update"]
69 if is_dict_field:
70 for item_key in command[key]:
71 command[key][item_key] = "%s"
72 continue
73  
74 # For pipeline fields strip the `$match` dict
75 is_pipeline_field = key == "pipeline"
76 if is_pipeline_field:
77 for pipeline in command[key]:
78 for match_key in pipeline["$match"] if "$match" in pipeline else []:
79 pipeline["$match"][match_key] = "%s"
80 continue
81  
82 # Default stripping
83 command[key] = "%s"
84  
85 return command
