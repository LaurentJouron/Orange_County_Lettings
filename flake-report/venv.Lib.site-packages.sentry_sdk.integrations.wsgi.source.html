160 def get_client_ip(environ):
161 # type: (Dict[str, str]) -> Optional[Any]
162 """
163 Infer the user IP address from various headers. This cannot be used in
164 security sensitive situations since the value may be forged from a client,
165 but it's good enough for the event payload.
166 """
167 try:
168 return environ["HTTP_X_FORWARDED_FOR"].split(",")[0].strip()
169 except (KeyError, IndexError):
170 pass

172 try:
173 return environ["HTTP_X_REAL_IP"]
174 except KeyError:
175 pass

180 def _capture_exception(hub):
181 # type: (Hub) -> ExcInfo
182 exc_info = sys.exc_info()
183  
184 # Check client here as it might have been unset while streaming response
185 if hub.client is not None:
186 e = exc_info[1]

235 def _make_wsgi_event_processor(environ, use_x_forwarded_for):
236 # type: (Dict[str, str], bool) -> EventProcessor
237 # It's a bit unfortunate that we have to extract and parse the request data
238 # from the environ so eagerly, but there are a few good reasons for this.
239 #
240 # We might be in a situation where the scope/hub never gets torn down
241 # properly. In that case we will have an unnecessary strong reference to
242 # all objects in the environ (some of which may take a lot of memory) when
243 # we're really just interested in a few of them.
244 #
245 # Keeping the environment around for longer than the request lifecycle is
246 # also not necessarily something uWSGI can deal with:
247 # https://github.com/unbit/uwsgi/issues/1950
248  
249 client_ip = get_client_ip(environ)
250 request_url = get_request_url(environ, use_x_forwarded_for)
251 query_string = environ.get("QUERY_STRING")
252 method = environ.get("REQUEST_METHOD")
253 env = dict(_get_environ(environ))
254 headers = _filter_headers(dict(_get_headers(environ)))
255  
256 def event_processor(event, hint):
257 # type: (Dict[str, Any], Dict[str, Any]) -> Dict[str, Any]
258 with capture_internal_exceptions():
259 # if the code below fails halfway through we at least have some data
260 request_info = event.setdefault("request", {})
261  
262 if _should_send_default_pii():
263 user_info = event.setdefault("user", {})
264 if client_ip:
265 user_info.setdefault("ip_address", client_ip)
266  
267 request_info["url"] = request_url
268 request_info["query_string"] = query_string
269 request_info["method"] = method
270 request_info["env"] = env
271 request_info["headers"] = headers
272  
273 return event
274  
275 return event_processor
