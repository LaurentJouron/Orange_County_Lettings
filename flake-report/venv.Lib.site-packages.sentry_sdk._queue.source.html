97 class Queue(object):
98 """Create a queue object with a given maximum size.
99  
100 If maxsize is <= 0, the queue size is infinite.
101 """
102  
103 def __init__(self, maxsize=0):
104 self.maxsize = maxsize
105 self._init(maxsize)
106  
107 # mutex must be held whenever the queue is mutating. All methods
108 # that acquire mutex must release it before returning. mutex
109 # is shared between the three conditions, so acquiring and
110 # releasing the conditions also acquires and releases mutex.
111 self.mutex = threading.RLock()
112  
113 # Notify not_empty whenever an item is added to the queue; a
114 # thread waiting to get is notified then.
115 self.not_empty = threading.Condition(self.mutex)
116  
117 # Notify not_full whenever an item is removed from the queue;
118 # a thread waiting to put is notified then.
119 self.not_full = threading.Condition(self.mutex)
120  
121 # Notify all_tasks_done whenever the number of unfinished tasks
122 # drops to zero; thread waiting to join() is notified to resume
123 self.all_tasks_done = threading.Condition(self.mutex)
124 self.unfinished_tasks = 0
125  
126 def task_done(self):
127 """Indicate that a formerly enqueued task is complete.
128  
129 Used by Queue consumer threads. For each get() used to fetch a task,
130 a subsequent call to task_done() tells the queue that the processing
131 on the task is complete.
132  
133 If a join() is currently blocking, it will resume when all items
134 have been processed (meaning that a task_done() call was received
135 for every item that had been put() into the queue).
136  
137 Raises a ValueError if called more times than there were items
138 placed in the queue.
139 """
140 with self.all_tasks_done:
141 unfinished = self.unfinished_tasks - 1
142 if unfinished <= 0:
143 if unfinished < 0:
144 raise ValueError("task_done() called too many times")
145 self.all_tasks_done.notify_all()
146 self.unfinished_tasks = unfinished
147  
148 def join(self):
149 """Blocks until all items in the Queue have been gotten and processed.
150  
151 The count of unfinished tasks goes up whenever an item is added to the
152 queue. The count goes down whenever a consumer thread calls task_done()
153 to indicate the item was retrieved and all work on it is complete.
154  
155 When the count of unfinished tasks drops to zero, join() unblocks.
156 """
157 with self.all_tasks_done:
158 while self.unfinished_tasks:
159 self.all_tasks_done.wait()
160  
161 def qsize(self):
162 """Return the approximate size of the queue (not reliable!)."""
163 with self.mutex:
164 return self._qsize()
165  
166 def empty(self):
167 """Return True if the queue is empty, False otherwise (not reliable!).
168  
169 This method is likely to be removed at some point. Use qsize() == 0
170 as a direct substitute, but be aware that either approach risks a race
171 condition where a queue can grow before the result of empty() or
172 qsize() can be used.
173  
174 To create code that needs to wait for all queued tasks to be
175 completed, the preferred technique is to use the join() method.
176 """
177 with self.mutex:
178 return not self._qsize()
179  
180 def full(self):
181 """Return True if the queue is full, False otherwise (not reliable!).
182  
183 This method is likely to be removed at some point. Use qsize() >= n
184 as a direct substitute, but be aware that either approach risks a race
185 condition where a queue can shrink before the result of full() or
186 qsize() can be used.
187 """
188 with self.mutex:
189 return 0 < self.maxsize <= self._qsize()
190  
191 def put(self, item, block=True, timeout=None):
192 """Put an item into the queue.
193  
194 If optional args 'block' is true and 'timeout' is None (the default),
195 block if necessary until a free slot is available. If 'timeout' is
196 a non-negative number, it blocks at most 'timeout' seconds and raises
197 the FullError exception if no free slot was available within that time.
198 Otherwise ('block' is false), put an item on the queue if a free slot
199 is immediately available, else raise the FullError exception ('timeout'
200 is ignored in that case).
201 """
202 with self.not_full:
203 if self.maxsize > 0:
204 if not block:
205 if self._qsize() >= self.maxsize:
206 raise FullError()
207 elif timeout is None:
208 while self._qsize() >= self.maxsize:
209 self.not_full.wait()
210 elif timeout < 0:
211 raise ValueError("'timeout' must be a non-negative number")
212 else:
213 endtime = time() + timeout
214 while self._qsize() >= self.maxsize:
215 remaining = endtime - time()
216 if remaining <= 0.0:
217 raise FullError()
218 self.not_full.wait(remaining)
219 self._put(item)
220 self.unfinished_tasks += 1
221 self.not_empty.notify()
222  
223 def get(self, block=True, timeout=None):
224 """Remove and return an item from the queue.
225  
226 If optional args 'block' is true and 'timeout' is None (the default),
227 block if necessary until an item is available. If 'timeout' is
228 a non-negative number, it blocks at most 'timeout' seconds and raises
229 the EmptyError exception if no item was available within that time.
230 Otherwise ('block' is false), return an item if one is immediately
231 available, else raise the EmptyError exception ('timeout' is ignored
232 in that case).
233 """
234 with self.not_empty:
235 if not block:
236 if not self._qsize():
237 raise EmptyError()
238 elif timeout is None:
239 while not self._qsize():
240 self.not_empty.wait()
241 elif timeout < 0:
242 raise ValueError("'timeout' must be a non-negative number")
243 else:
244 endtime = time() + timeout
245 while not self._qsize():
246 remaining = endtime - time()
247 if remaining <= 0.0:
248 raise EmptyError()
249 self.not_empty.wait(remaining)
250 item = self._get()
251 self.not_full.notify()
252 return item
253  
254 def put_nowait(self, item):
255 """Put an item into the queue without blocking.
256  
257 Only enqueue the item if a free slot is immediately available.
258 Otherwise raise the FullError exception.
259 """
260 return self.put(item, block=False)
261  
262 def get_nowait(self):
263 """Remove and return an item from the queue without blocking.
264  
265 Only get an item if one is immediately available. Otherwise
266 raise the EmptyError exception.
267 """
268 return self.get(block=False)

282 def _put(self, item):
